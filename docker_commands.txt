create docker image:
    docker build -t rest-apis-flask-python .
 
spin up container from image:
    docker run -p 5000:5000 rest-apis-flask-python
 
spin up container leveraging and auto-update:
    docker run -dp 5000:5000 -w /app -v "$(pwd):/app" rest-apis-flask-python
 
    (this will output the container ID)
 
run docker container locally after leveraging gunicorn:
    docker run -dp 5000:5000 -w /app -v "$(pwd):/app" rest-apis-flask-python sh -c "flask run --host 0.0.0.0"
 
list running containers:
    docker ps
 
stopping containers:
    docker stop CONTAINER_ID/NAME
 
remove container:
    docker rm CONTAINER_ID/NAME
